Easy-Swift-2

2019-09-11 swift

类是模具，对象是铸件。

Class

class Cat {
	var name: String?
	var age: Int?
}

var kitty = Cat()
kitty.name = "Hello Kitty"
kitty.age = 45

Method

class Cat {

    func sayHello(name:String) -> Bool {
		print("Hello, my name is \(name)")		
		return true
	}
}

var kitty = Cat()
kitty.sayHello(name: "Kitty")
