Easy-Ruby

2019-09-03 Ruby

A dynamic, open source programming language with a focus on simplicity and productivity. It has an elegant syntax that is natural to read and easy to write.

1. Constant & Variable

my_name = "小明" // Variable
MyName = "小明" // Constant

2. Control Flow

day = 5

if day==0
	puts "星期日"
elsif day==6
	puts "星期六"
else
	puts "工作日"
end

3. Data Type

  • Number
  • String
  • Range
  • Array
  • Hash
  • Symbol

类型

puts 5.class  # => Integer

数字

puts 10 + 3    # => 13
puts 10 - 3    # => 7
puts 10 * 3    # => 30
puts 10 / 3    # => 3
puts 10.0 / 3  # => 3.3333333333333335
puts 10 / 3.0  # => 3.3333333333333335
puts 10 % 3    # => 1

字符串

name = "小明"
age = 18

puts "你好，我是" + name + "，我今年" + age.to_s + "岁" # => 你好，我是小明，我今年18岁

puts "你好，我是#{name}，我今年#{age}岁" # => 你好，我是小明，我今年18岁

范围

(1..5).to_a   # 1 到 5
(1...5).to_a  # 1 到 4

数组

thoughts = ["儒家", "道家", "墨家", "法家"]
p thoughts[1] # => 道家
p thoughts[-1] # => 法家
p thoughts.first # => 儒家
p thoughts.last # => 法家

字典

thought_hash = {

puts thought_hash.values # => []

3. Methods

	return "hello, #{name}"

4. Block

def say_hello
	puts "开始"
	yield 123        # 把控制权暂时让给 Block
	puts "结束"
end

say_hello { |x|
	puts "这里是 Block，我收到了#{x}"
}

5. Class

class Cat
  attr_accessor :name #属性 

  def initialize(name, gender)
    @name = name # 实例变量
    @gender = gender
  end
    
  def say_hello
    puts "hello, my name is #{@name}" # => hello, my name is kitty
  end
end

6. Module

module Flyable
  def fly
    puts "I can fly!"
  end
end

class Cat
  include Flyable
end

kitty = Cat.new
kitty.fly        # => I can fly!
