//模拟发送http请求
const request = require("request");

let addressArr = ["广州市" , "中山市", "佛山市"]


for (let index = 0; index < addressArr.length; index++) {
  const address = addressArr[index];
  let url = "https://apis.map.qq.com/jsapi?qt=geoc&addr=" + address + "&output=jsonp&pf=jsapi&ref=jsapi"//&cb=qq.maps._svcb2.geocoder0"

  //get请求
  console.log(encodeURI(url));
  request(encodeURI(url), function (error, response, body) {
    if (!error && response.statusCode == 200) {
      // console.log(JSON.parse(body))
      let j = JSON.parse(body)
      console.log(address + "-" + j.detail.pointx + "-" + j.detail.pointy)
    }
    else {
      console.log(error);
    }
  });
}
